#include <iostream>
using namespace std;

int main(){
     int rows, columns, firstList[100][100], secondList[100][100], sumList[100][100], i, j;

     cout << "Row number? (<100)";
     cin >> rows;
     cout << "Column number? (<100)";
     cin >> columns;

     cout << endl << "Items of the first Matrix: " << endl;
     for(i = 0; i < rows; ++i){
          for(j = 0; j < columns; ++j){
               cout << "First Matrix:" << i + 1 << j + 1 << " : ";
               cin >> firstList[i][j];
          }
     }

     cout << endl << "Items of the second Matrix: " << endl;
     for(i = 0; i < rows; ++i){
          for(j = 0; j < columns; ++j){
               cout << "Second Matrix:" << i + 1 << j + 1 << " : ";
               cin >> secondList[i][j];
          }
     }

     // Toplama
     for(i = 0; i < rows; ++i){
          for(j = 0; j < columns; ++j){
               sumList[i][j] = firstList[i][j] + secondList[i][j];
          }
     }
     // Sonuclar
     cout << endl << "SUM Matrix: " << endl;
     for(i = 0; i < rows; ++i){
          for(j = 0; j < columns; ++j){
               cout << sumList[i][j] << " ";
               if(j == columns - 1){
                    cout << endl;
               }
          }
     }
     return 0;
}
