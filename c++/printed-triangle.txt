#include<iostream>
using namespace std;

int main(){
	int a=9;

	cout << "\"UpsideDown Triangle\":\n\n";

	for(int i=1; i<=5; i++){
		for(int k=0; k<a; k++){
			cout << "*"; // Print an asterix sign
		}

  	      	cout << endl;
  	      	a = a - 2;

     		for(int j=0; j<i; j++){
			cout << " "; // Print an empty space
        	}
	}
	return 0;
}
